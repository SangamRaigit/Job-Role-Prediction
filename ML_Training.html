def log_classifier(features,target):
    """Logistic Regression with parameter values providing best prossible accuracy
    random_state: random_state is the seed used by the random number generator,
    Maximum number of iterations taken for the solvers to converge."""
    logReg = LogisticRegression(random_state=46,max_iter = 200)
    logReg.fit(features,target)
    return logReg

def rf_classifier(features,target):
    """Random Forest Classifier 
    n_estimators: list of DecisionTreeClassifier,
    random_state: random_state is the seed used by the random number generator,
    min_samples_leaf:The minimum number of samples required to be at a leaf node """
    clf = RandomForestClassifier(n_estimators = 20, min_samples_leaf = 20, random_state = 46)
    clf.fit(features,target)
    return clf

def gbm_classifier(features,target):
    """This is gradient boosting model classifier.Decision trees are usually used when doing gradient boosting.
    n_estimators: The number of boosting stages to perform.
    random_state: random_state is the seed used by the random number generator.
    max_depth:The maximum depth limits the number of nodes in the tree.
    """
    clf = GradientBoostingRegressor(n_estimators = 30)
    clf.fit(features,target)
    return clf	

def nn_classifier(features,target):
    """Neural Net Classifier 
    MLP with hidden layers have a non-convex loss function where there exists more than one local minimum,
    random_state: random_state is the seed used by the random number generator,
    Maximum number of iterations taken for the solvers to converge.
    """
    clf = MLPClassifier(hidden_layer_sizes = (40,40,40),max_iter=501,random_state=546)
    clf.fit(features,target)
    return clf

def print_all_accuracy_matric(algorithm,name):
    #computing the accuracy, f1_score, Recall, precision of the model performance
    #predicting the target value from the model for the samples
    y_train_log = algorithm.predict(x_train)
    y_test_log = algorithm.predict(x_val)

    acc_train_log = metrics.accuracy_score(y_train,y_train_log)
    acc_test_log = metrics.accuracy_score(y_val,y_test_log)
    print(name," : Accuracy on training Data: {:.3f}".format(acc_train_log))
    print(name," : Accuracy on test Data: {:.3f}".format(acc_test_log))
    print()

    f1_score_train_log = metrics.f1_score(y_train,y_train_log, average='macro')
    f1_score_test_log = metrics.f1_score(y_val,y_test_log, average='macro')
    print(name," : f1_score on training Data: {:.3f}".format(f1_score_train_log))
    print(name," : f1_score on test Data: {:.3f}".format(f1_score_test_log))
    print()

    recall_score_train_log = metrics.recall_score(y_train,y_train_log, average='macro')
    recall_score_test_log = metrics.recall_score(y_val,y_test_log, average='macro')
    print(name," : Recall on training Data: {:.3f}".format(recall_score_train_log))
    print(name," : Recall on test Data: {:.3f}".format(recall_score_test_log))
    print()

    precision_score_train_log = metrics.precision_score(y_train,y_train_log, average='macro')
    precision_score_test_log = metrics.precision_score(y_val,y_test_log, average='macro')
    print(name," : precision on training Data: {:.3f}".format(precision_score_train_log))
    print(name," : precision on test Data: {:.3f}".format(precision_score_test_log))

    print('\n\nClassification Report\n\n')
    print(metrics.classification_report(y_val, y_test_log))
    print('\n\n')
